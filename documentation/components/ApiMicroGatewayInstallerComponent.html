import { Component, ChangeDetectorRef, Input,
			OnInit, OnDestroy, ViewChild }   	from '@angular/core';
import { FormBuilder, FormGroup, FormControl }  from '@angular/forms';

import { BuilderComponent, BuilderProperties,
				PropertiesChangedOwner }		from '../elements/build-image-choose.directive';

import { ResourceService }                      from '../../services/resources.service';
import { BuildParams, BuildCommand }     		from '../../models/project';

@Component({
  selector: 'jc-apimg-install',
  templateUrl: '../../templates/plugins/apimg-install.html'
})
export class ApiMicroGatewayInstallerComponent implements BuilderComponent {

	public formGroup: FormGroup;
	public owner: PropertiesChangedOwner;
	public dataSource: BuildParams;

	public primaryPortCtrl: FormControl;
	public httpsPortCtrl: FormControl;
	public apiGatewayUrlCtrl: FormControl;
	public apiGatewayUserCtrl: FormControl;
	public apiGatewayPasswordCtrl: FormControl;
	public defaultApis: FormControl;
	public licCtrl: FormControl;

	public isInstall: boolean = false;

  	public licenceFiles: string[] = [];

	public constructor(private _resources: ResourceService) {
		this._resources.resourcesForType("licenses").subscribe((p) => {
        this.licenceFiles = [];
         p.forEach((f) => {
          this.licenceFiles.push(f.name);
        })
      });
	}

	public ngOnInit() {

		this.dataSource = this.owner.initialProperties();

		if ((<any> this.dataSource).productType) {
			this.isInstall = true;
			this.setDefaultsForInstallation();
			this.setupValuesForInstallation();
		}
		else {
			this.setDefaultsForBuilding();
			this.setupValuesForBuilding();
		}
	}

		public refreshBuildCommands(show: boolean) {

  }

	public setLicenseFile(license: string) {

		let file: BuildCommand = this.dataSource.fileForType("licenses");

	    if (!file) {
	        file = new BuildCommand();
	        file.commandType = "file"
	        file.fileType = "licenses";
	        file.description = "License file"
	        file.target = "/opt/softwareag/Microgateway/config/license.xml";

	        //ensures that the build-components view gets updated
	        this.dataSource.buildCommands = Object.assign([], this.dataSource.buildCommands);

	        this.dataSource.buildCommands.push(file);
	    }

      	file.source = license.replace(/\s/g, "-");
	}

	private setupValuesForInstallation() {

		this.setupValuesForBuilding();

		if (this.dataSource.publicPort())
			this.primaryPortCtrl.disable();

		this.licCtrl = new FormControl(this.currentAPIGatewayLicense());
		this.formGroup.addControl("licCtrl", this.licCtrl);
	}

	private setupValuesForBuilding() {

		this.primaryPortCtrl = new FormControl(this.dataSource.publicPort());
		this.httpsPortCtrl = new FormControl(this.dataSource.args.get('mcgw_ports_https'));
		this.apiGatewayUrlCtrl = new FormControl(this.dataSource.args.get("mcgw_api_gateway_url"));
		this.apiGatewayUserCtrl = new FormControl(this.dataSource.args.get("mcgw_api_gateway_user"));
		this.apiGatewayPasswordCtrl = new FormControl(this.dataSource.args.get("mcgw_api_gateway_password"));

		this.formGroup.addControl("primaryPortCtrl", this.primaryPortCtrl);
		this.formGroup.addControl("httpsPortCtrl", this.httpsPortCtrl);
		this.formGroup.addControl("apiGatewayUrlCtrl", this.apiGatewayUrlCtrl);
		this.formGroup.addControl("apiGatewayUserCtrl", this.apiGatewayUserCtrl);
		this.formGroup.addControl("apiGatewayPasswordCtrl", this.apiGatewayPasswordCtrl);

		this.flagChanges(this.primaryPortCtrl.value);

		this.formGroup.valueChanges.subscribe((v) => {

			if (this.primaryPortCtrl && this.primaryPortCtrl.dirty) {
				this.dataSource.setPublicPort(this.primaryPortCtrl.value);
				this.dataSource.args.set('mcgw_ports_http', this.primaryPortCtrl.value);
				this.primaryPortCtrl.markAsPristine();
			}

			if (this.httpsPortCtrl && this.httpsPortCtrl.dirty) {
			this.dataSource.args.set('mcgw_ports_https', this.httpsPortCtrl.value);
				this.httpsPortCtrl.markAsPristine();
			}

			if (this.licCtrl && this.licCtrl.dirty) {
				this.setAPIGatewayLicense(this.licCtrl.value);
				this.licCtrl.markAsPristine();
			}

			if (this.apiGatewayUrlCtrl.dirty) {

				this.dataSource.args.set("mcgw_api_gateway_url", this.apiGatewayUrlCtrl.value);
				this.apiGatewayUrlCtrl.markAsPristine();
			}

			if (this.apiGatewayUserCtrl.dirty) {

				this.dataSource.args.set("mcgw_api_gateway_user", this.apiGatewayUserCtrl.value);
				this.apiGatewayUserCtrl.markAsPristine();
			}

			if (this.apiGatewayPasswordCtrl.dirty) {

				this.dataSource.args.set("mcgw_api_gateway_password", this.apiGatewayPasswordCtrl.value);
				this.apiGatewayPasswordCtrl.markAsPristine();
			}

			this.flagChanges(this.primaryPortCtrl.value)
		});
	}

	private setDefaultsForInstallation() {

      	let yum: BuildCommand = this.dataSource.fileForType("yum");

		if (!yum) {

		    // add if not already in list

		    yum = new BuildCommand();
		    yum.commandType = "run";
		    yum.buildTarget = "pre-install";
		    yum.fileType = "yum";
		    yum.description = "Install unzip so that we can extract microgateway build below";
		    yum.target = "yum install -y unzip";
		    this.dataSource.buildCommands.push(yum);

		    let create: BuildCommand = new BuildCommand();
		    create.commandType = "run";
		    create.buildTarget = "post-install";
		    create.fileType = "microgateway.sh";
		    create.description = "Script to generate micro gateway zip file";
		    create.target = "cd /opt/softwareag/Microgateway; ./microgateway.sh createInstance -os linux -instance /opt/softwareag/microgateway.zip; cd ..;unzip microgateway.zip -d microgateway-linux";
		    this.dataSource.buildCommands.push(create);

			let extract: BuildCommand = new BuildCommand();
		    extract.commandType = "copy";
		    extract.buildTarget = "build";
		    extract.fileType = "copy";
		    extract.description = "copy from install build to build target";
		    extract.source = "/opt/softwareag/microgateway-linux";
		    extract.target = "/opt/softwareag/Microgateway";
		    this.dataSource.buildCommands.push(extract);

			let configure: BuildCommand = new BuildCommand();
		    configure.commandType = "run";
		    configure.buildTarget = "build";
		    configure.fileType = "configure";
		    configure.description = "set permissions";
		    configure.target = "cd /opt/softwareag/Microgateway; chmod u+x ./setpermissions.sh; /opt/softwareag/Microgateway/setpermissions.sh";
		    this.dataSource.buildCommands.push(configure);

		    this.setDefaultsForBuilding();

		    //ensures that the build-components view gets updated
		    this.dataSource.buildCommands = Object.assign([], this.dataSource.buildCommands);
		}
  	}

  	private setDefaultsForBuilding() {

  		let settings: BuildCommand = this.dataSource.fileForType("resource");

  		if (!settings) {

  			settings = new BuildCommand();
		    settings.commandType = "file";
		    settings.buildTarget = "build";
		    settings.fileType = "resource";
		    settings.description = "system settings for micro gateway";
		    settings.source = "mcgw-system-settings.yml";
		    settings.target = "/opt/softwareag/Microgateway/config/system-settings.yml";
		    this.dataSource.buildCommands.push(settings);

		    settings = new BuildCommand();
		    settings.commandType = "file";
		    settings.buildTarget = "build";
		    settings.fileType = "resource";
		    settings.description = "custom settings for micro gateway";
		    settings.source = "mcgw-custom-settings.yml";
		    settings.target = "/opt/softwareag/Microgateway/config/custom-settings.yml";
		    this.dataSource.buildCommands.push(settings);

		    this.dataSource.args.set("env_mcgw-custom-settings", "custom-settings.yml");

		    this.dataSource.buildCommands = Object.assign([], this.dataSource.buildCommands);

		    if (!this.isInstall) {

		    	if (!this.dataSource.isCustomImage()) {

		    		let init: BuildCommand = new BuildCommand();
				    init.commandType = "file";
				    init.buildTarget = "build";
				    init.fileType = "support";
				    init.description = "wrapper for docker init";
				    init.source = "apimg-docker-init.sh";
				    init.target = "/docker-init.sh";
				    this.dataSource.buildCommands.push(init);
				    this.dataSource.entryPoint = "/docker-init.sh";
		    	}

			    this.dataSource.args.set("env_mcgw-custom-settings", "custom-settings.yml");

			    this.dataSource.buildCommands = Object.assign([], this.dataSource.buildCommands);

		    }
		}
  	}

  	private currentAPIGatewayLicense(): string {

  		let lic: BuildCommand = this.dataSource.fileForType("licenses");

  		if (lic) {
  			return lic.source;
  		} else {
  			return null;
  		}
  	}

  	private setAPIGatewayLicense(license: string) {

  		let lic: BuildCommand = this.dataSource.fileForType("licenses");

		if (!lic) {

		    // add if not already in list

		    lic = new BuildCommand();
		    lic.commandType = "file";
		    lic.buildTarget = "install";
		    lic.fileType = "licenses";
		    lic.description = "License file for AP Gateway";
		    lic.target = "/software";

		    this.dataSource.buildCommands.push(lic);
		    this.dataSource.buildCommands = Object.assign([], this.dataSource.buildCommands);
		}

		lic.source = license.replace(/\s/g, "-")
		this.dataSource.args.set("apiGatewayLicenseFile", lic.source);

		 //ensures that the build-components view gets updated

		let altLic: BuildCommand = this.dataSource.fileForType("lic");

		if (!altLic) {

			altLic = new BuildCommand();
		    altLic.commandType = "file";
		    altLic.buildTarget = "build";
		    altLic.fileType = "lic";
		    altLic.description = "copy micro gateway license to runtime";
		    altLic.target = "/opt/softwareag/Microgateway/config/license.xml";

		    this.dataSource.buildCommands.push(altLic);
		    this.dataSource.buildCommands = Object.assign([], this.dataSource.buildCommands);
		}

		altLic.source = "/resources/" + this.dataSource.assignedLicense() + ".xml";
  	}

	private flagChanges(port: string) {

		let props: Map<string, BuilderProperties[]> = new Map();

    	if (port)
		  props.set("API Entry Port", [new BuilderProperties("HTTP", [port])]);

		this.owner.propertiesChangedInBuilder("micro-gateway", props, this.dataSource);
	}
}
