@Component({
  selector: 'test-case',
  templateUrl: '../templates/test-case.html',
  styleUrls: ['../templates/test-case.css']
})

export class TestCaseComponent implements OnInit {

	@Input()
	public package: WmPackageInfo;

	@Input()
	public testCase: TestCase;

	@Input()
	public form: FormGroup;

	@Output()
	public testCaseChanged: EventEmitter<TestCase> = new EventEmitter();

	public packageCtrl: FormControl;
	public testTypeCtrl: FormControl;
	public serviceCtrl: FormControl;
	public apiCtrl: FormControl;

	public packages: WmPackageInfo[];

	public constructor(private _formBuilder: FormBuilder, private _testConfig: TestConfigService) {

		_testConfig.packages().subscribe((p) => {
			this.packages = p;
		});
	}

	public ngOnInit() {

		this.packageCtrl = new FormControl(this.packageForValue(this.package.name));
		this.serviceCtrl = new FormControl(this.testCase.endpoint);
		this.testTypeCtrl = new FormControl(this.testCase.api ? "api" : "service");
		this.apiCtrl = new FormControl(this.apiForValue(this.testCase.api));

		this.form.addControl("packageCtrl", this.packageCtrl);
		this.form.addControl("testTypeCtrl", this.testTypeCtrl);
		this.form.addControl("serviceCtrl", this.serviceCtrl);
		this.form.addControl("apiCtrl", this.apiCtrl);

		this.form.valueChanges.subscribe((f) => {

			if (this.packageCtrl.dirty) {
				this.package = this.packageCtrl.value;
				this.packageCtrl.markAsPristine();
			}
		});
	}

	public services(): string[] {

		if (this.package) {
			return this.package.services;
		} else {
			return [];
		}
	}

	public apis(): any[] {

		if (this.package) {
			return this.package.apis;
		} else {
			return [];
		}
	}

	public packageForValue(value: string): any {

		let found: any = null;

		this.packages.forEach((p) => {

			if (p.name == value)
				found = p;
		});

		return found;
	}

	public apiForValue(value: string): any {

		let found: any = null;

		this.package.apis.forEach((a) => {

			if (a.swaggerEndPoint.indexOf(value) != -1)
				found = a;
		});

		return found;
	}
}
