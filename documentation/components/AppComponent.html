import { Component, ChangeDetectorRef }                from '@angular/core';

import { RouterModule, ActivatedRoute, Routes, RouterOutlet, 
                                      Route, Router }  from '@angular/router';

export class AppComponent implements OnInit, OnDestroy {
  
  mobileQuery: MediaQueryList;

  private _mobileQueryListener: () => void;

  public static linksFunc: Function;
  public static labelForPathFunc: Function;
  public static iconForPathFunc: Function;
  public static iconClassForPathFunc: Function;
  public static childrenForPathFunc: Function;

  public currentParentMenu: string;

  public sideNavColor: any = {color: "orange"};

  @ViewChild(MatAccordion) 
  navList: MatAccordion;

 @ViewChild('snav') 
 sidebar: MatDrawer;

  constructor(private _location: Location, changeDetectorRef: ChangeDetectorRef, media: MediaMatcher, private _dialog: MatDialog, private _router: Router, private _route: ActivatedRoute) {
    
    this.mobileQuery = media.matchMedia('(max-width: 600px)');
    this._mobileQueryListener = () => changeDetectorRef.detectChanges();
    this.mobileQuery.addListener(this._mobileQueryListener);
  }

  public ngOnInit() {
    
    this.navList.multi = true;
    this.currentParentMenu = this._route['_routerState'].snapshot.url.substring(1) || "runtime";

    if (this._location.path() == "") {
      let ref: AppComponent = this;
      setTimeout(() => {
        ref.sidebar.open();
      }, 400);
    }
  }

  public ngOnDestroy() {

    this.mobileQuery.removeListener(this._mobileQueryListener);
  }

  public sideBarOpening() {

     this.sideNavColor = {color: "orange"};
  }

  public sideBarClosing() {

     this.sideNavColor = {color: "white"};
  }

  public linksForLink(link: string): string[] {

    if (AppComponent.linksFunc)
      return AppComponent.linksFunc(link);
    else
      return [];
  }

  public parentMenuExpanded(path: string) {

    this.currentParentMenu = path;
  }

  public parentMenuCollapsed(path: string) {

    this.currentParentMenu = null;
  }

  public selectParentMenu(path: string) {

    //this.currentParentMenu = path;
    this._router.navigate(['/' + path], { skipLocationChange: false });
  }

  public selectSubMenu(path: any) {

    // hide any expanded menu items

     this._router.navigate(['/' + path], { skipLocationChange: false });
  }
  public isSelected(path: string): boolean {

    if ("/" + path == this._route['_routerState'].snapshot.url)
      return true;
    else
      return false;
  }

  public hasChildren(path: string): boolean {

    if (AppComponent.childrenForPathFunc) {

      let x: string[] = AppComponent.childrenForPathFunc(path);

      return x.length > 0;
    }
    else {
      return false;  
    }
  }

  public prepareRoute(outlet: RouterOutlet): boolean {
    return outlet && outlet.activatedRouteData && outlet.activatedRouteData['animation'];
  }

  public labelForPath(path: string) {

    if (AppComponent.labelForPathFunc)
      return AppComponent.labelForPathFunc(path);
    else
      return path;
  }

  public isImage(path: string): boolean {

    if (AppComponent.iconClassForPathFunc)
      return AppComponent.iconClassForPathFunc(path) == "img";
    else
      return false;
  }

  public iconClassForPath(path: string): string {

    if (AppComponent.iconClassForPathFunc)
      return AppComponent.iconClassForPathFunc(path);
    else
      return path;
  }

  public iconForPath(path: string): string {

    if (AppComponent.iconForPathFunc)
      return AppComponent.iconForPathFunc(path);
    else
      return path;
  }

  public showAboutPanel() {

    let dialogRef = this._dialog.open(AboutComponent, {
      height: '400px',
      width: '600px',
    });
  }

  private isPathAssociatedWithParent(parent: string, path: string): boolean {

    let children: string[] = AppComponent.childrenForPathFunc(parent);

    return children && children.indexOf(path) != -1;
  }
}
