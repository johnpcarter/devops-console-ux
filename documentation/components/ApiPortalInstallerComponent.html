import { Component, ChangeDetectorRef, Input,
			OnInit, OnDestroy, ViewChild }   	from '@angular/core';
import { FormBuilder, FormGroup, FormControl }  from '@angular/forms';

import { BuilderComponent, BuilderProperties,
				PropertiesChangedOwner }		from '../elements/build-image-choose.directive';

import { BuildParams, BuildCommand }       		from '../../models/project';

@Component({
  selector: 'jc-apipr-installer',
  templateUrl: '../../templates/plugins/apipr-install.html'
})

export class ApiPortalInstallerComponent implements BuilderComponent {

	public formGroup: FormGroup;
	public owner: PropertiesChangedOwner;
	public dataSource: BuildParams;

	public primaryPortCtrl: FormControl;
	public securePortCtrl: FormControl;

	public ngOnInit() {

		this.dataSource = this.owner.initialProperties();

		this.setDefaults();

		this.primaryPortCtrl = new FormControl(this.dataSource.publicPort());
		this.securePortCtrl = new FormControl(this.dataSource.args.get("securePort"));

		this.primaryPortCtrl.disable();

		this.formGroup.addControl("primaryPortCtrl", this.primaryPortCtrl);
		this.formGroup.addControl("securePortCtrl", this.securePortCtrl);

		this.flagChanges(this.primaryPortCtrl.value, this.securePortCtrl.value);

		this.formGroup.valueChanges.subscribe((v) => {
			this.dataSource.args.set('securePort', this.securePortCtrl.value);
			this.flagChanges(this.primaryPortCtrl.value, this.securePortCtrl.value)
		});
	}

		public refreshBuildCommands(show: boolean) {

  }

	public setLicenseFile(license: string) {

		let file: BuildCommand = this.dataSource.fileForType("licenses");

	    if (!file) {
	        file = new BuildCommand();
	        file.commandType = "file"
	        file.fileType = "licenses";
	        file.description = "License file"
	        file.target = "/opt/softwareag/API_Portal/license.xml";

	        //ensures that the build-components view gets updated
	        this.dataSource.buildCommands = Object.assign([], this.dataSource.buildCommands);

	        this.dataSource.buildCommands.push(file);
	    }

      	file.source = license.replace(/\s/g, "-");
	}

	private setDefaults() {

		if (!this.dataSource.args.get("securePort")) {
			this.dataSource.args.set("securePort", "18102");
		}

		this.dataSource.hostName = "api-portal";

		let yum: BuildCommand = this.dataSource.fileForType("yum");

		if (!yum) {

		 	yum = new BuildCommand();
		    yum.commandType = "run";
		    yum.buildTarget = "pre-build";
		    yum.fileType = "yum";
		    yum.description = "Required in order for Portal to start";
		    yum.target = "yum install -y fontconfig";
		    this.dataSource.buildCommands.push(yum);

		    let config = new BuildCommand();
		    config.commandType = "file";
		    config.buildTarget = "build";
		    config.fileType = "install";
		    config.description = "Ensures that all services are started automatically when starting up";
		    config.source = "agent.user.cfg";
		    config.target = "/opt/softwareag/API_Portal/server/bin/agent.user.cfg";
		    this.dataSource.buildCommands.push(config);
		}
	}

	private flagChanges(port: string, securePort: string) {

		let props: Map<string, BuilderProperties[]> = new Map();

    	if (port)
		  props.set("UI Ports", [new BuilderProperties("HTTP", [port]), new BuilderProperties("HTTPs", [securePort])]);

		this.owner.propertiesChangedInBuilder("api-portal", props, this.dataSource);
	}
}
