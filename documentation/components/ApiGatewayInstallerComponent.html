import { Component, ChangeDetectorRef, Input,
			OnInit, OnDestroy, ViewChild }   	from '@angular/core';
import { FormBuilder, FormGroup, FormControl }  from '@angular/forms';

import { BuilderComponent, BuilderProperties,
				PropertiesChangedOwner }		from '../elements/build-image-choose.directive';

import { BuildParams, BuildCommand }       		from '../../models/project';

@Component({
  selector: 'jc-apigw-install',
  templateUrl: '../../templates/plugins/apigw-install.html'
})
export class ApiGatewayInstallerComponent implements BuilderComponent {

	public formGroup: FormGroup;
	public owner: PropertiesChangedOwner;
	public dataSource: BuildParams;

	public primaryPortCtrl: FormControl;
	public uIPortCtrl: FormControl;
	public secureUIPortCtrl: FormControl;

	public ElkPortHttpCtrl: FormControl;
	public ElkPortTcpCtrl: FormControl;

	public ngOnInit() {

		this.dataSource = this.owner.initialProperties();

		this.setDefaults();

		this.primaryPortCtrl = new FormControl(this.dataSource.publicPort());
		this.uIPortCtrl = new FormControl(this.dataSource.args.get("uIPort"));
		this.secureUIPortCtrl = new FormControl(this.dataSource.args.get("uIPortSecure"));

		this.ElkPortHttpCtrl = new FormControl(this.dataSource.args.get("ElkPortHttp"));
		this.ElkPortTcpCtrl = new FormControl(this.dataSource.args.get("ElkPortTCP"));

		this.primaryPortCtrl.disable();

		this.formGroup.addControl("primaryPortCtrl", this.primaryPortCtrl);
		this.formGroup.addControl("uIPortCtrl", this.uIPortCtrl);
		this.formGroup.addControl("secureUIPortCtrl", this.secureUIPortCtrl);
		this.formGroup.addControl("ElkPortHttpCtrl", this.ElkPortHttpCtrl);
		this.formGroup.addControl("ElkPortTcpCtrl", this.ElkPortTcpCtrl);

		this.flagChanges(this.uIPortCtrl.value, this.secureUIPortCtrl.value);

		this.formGroup.valueChanges.subscribe((v) => {
			this.dataSource.args.set('uIPort', this.uIPortCtrl.value);
			this.dataSource.args.set('uIPortSecure', this.secureUIPortCtrl.value);
			this.dataSource.args.set('ElkPortHttp', this.ElkPortHttpCtrl.value);
			this.dataSource.args.set('ElkPortTCP', this.ElkPortTcpCtrl.value);

			this.flagChanges(this.uIPortCtrl.value, this.secureUIPortCtrl.value)
		});
	}

	public refreshBuildCommands(show: boolean) {

  }

	public setLicenseFile(license: string) {

		let file: BuildCommand = this.dataSource.fileForType("licenses");

	    if (!file) {
	        file = new BuildCommand();
	        file.commandType = "file"
	        file.fileType = "licenses";
	        file.description = "License file"
	        file.target = "/opt/softwareag/IntegrationServer/instances/default/config/licenseKey.xml";

	        //ensures that the build-components view gets updated
	        this.dataSource.buildCommands = Object.assign([], this.dataSource.buildCommands);

	        this.dataSource.buildCommands.push(file);
	    }

      	file.source = license.replace(/\s/g, "-");
	}

	private setDefaults() {

		if (!this.dataSource.args.get("uIPort")) {
			this.dataSource.args.set("uIPort", "9072");
		}

		if (!this.dataSource.args.get("uIPortSecure")) {
			this.dataSource.args.set("uIPortSecure", "9073");
		}

		if (!this.dataSource.args.get("ElkPortHttp")) {
			this.dataSource.args.set("ElkPortHttp", "2405");
		}

		if (!this.dataSource.args.get("ElkPortTCP")) {
			this.dataSource.args.set("ElkPortTCP", "9340");
		}

		// Hidden properties

		let db: BuildCommand = this.dataSource.fileForType("isdb");

      	if (!db) {
        	db = new BuildCommand();
	        db.commandType = "run";
	        db.buildTarget = "build";
	        db.fileType = "isdb";
	        db.target = "mkdir /opt/softwareag/IntegrationServer/instances/default/db";
	        this.dataSource.buildCommands.push(db);
	    }

	    let cache: BuildCommand = this.dataSource.fileForType("cache");

	    if (!cache) {
	        cache = new BuildCommand();
	        cache.commandType = "run";
	        cache.buildTarget = "build";
	        cache.fileType = "cache";
	        cache.target = "mkdir /opt/softwareag/IntegrationServer/instances/default/cache";
	        this.dataSource.buildCommands.push(cache);
	    }

	   let eventstore: BuildCommand = this.dataSource.fileForType("eventstore");

	    if (!eventstore) {
	        eventstore = new BuildCommand();
	        eventstore.commandType = "run";
	        eventstore.buildTarget = "build";
	        eventstore.fileType = "eventstore";
	        eventstore.target = "mkdir -p /opt/softwareag/EventDataStore/data";
	        this.dataSource.buildCommands.push(eventstore);
	    }

	    let internal: BuildCommand = this.dataSource.fileForType("internal");

	    if (!internal) {
	        internal = new BuildCommand();
	        internal.commandType = "run";
	        internal.buildTarget = "build";
	        internal.fileType = "internal";
	        internal.target = "mkdir -p /opt/softwareag/InternalDataStore/data";
	        this.dataSource.buildCommands.push(internal);
	    }
	}

	private flagChanges(port: string, securePort: string) {

		let props: Map<string, BuilderProperties[]> = new Map();

    	if (port)
		  props.set("UI Ports", [new BuilderProperties("HTTP", [port]), new BuilderProperties("HTTPs", [securePort])]);

		this.owner.propertiesChangedInBuilder("api-gateway", props, this.dataSource);
	}
}
